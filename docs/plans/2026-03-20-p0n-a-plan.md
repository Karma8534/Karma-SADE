**Goal:** Add a `/cc` proxy route to hub-bridge that forwards requests to a CC server running on P1 (Tailscale 100.124.194.102), so Colby can communicate with CC at hub.arknexus.net/cc from any browser.

**Architecture:** A lightweight HTTP server runs on P1 (port 7891) accepting POST JSON, spawning `claude -p "message" --resume` subprocess, returning the response. hub-bridge adds a `/cc` route that authenticates with the existing Bearer token, then proxies to P1:7891 via Tailscale. **Tech Stack:** Node.js (hub-bridge server.js), Python (P1 CC server), Tailscale networking, existing Bearer token auth.

```python
#!/usr/bin/env python3
"""
P0N-A: CC persistent server on P1.
Accepts POST /cc with JSON {message, session_id?}
Runs: claude -p "message" --resume
Returns: {response, session_id, ok}
Auth: Bearer token checked against HUB_CHAT_TOKEN env var.
"""
import os, subprocess, json
from http.server import HTTPServer, BaseHTTPRequestHandler

PORT = 7891
TOKEN = os.environ.get("HUB_CHAT_TOKEN", "")

class CCHandler(BaseHTTPRequestHandler):
    def log_message(self, format, *args):
        print(f"[cc-server] {format % args}")

    def do_GET(self):
        if self.path == "/health":
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"ok": True, "service": "cc-server-p1"}).encode())
        else:
            self.send_response(404)
            self.end_headers()

    def do_POST(self):
        # Auth check
        auth = self.headers.get("Authorization", "")
        if TOKEN and auth != f"Bearer {TOKEN}":
            self.send_response(401)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"ok": False, "error": "Unauthorized"}).encode())
            return

        if self.path != "/cc":
            self.send_response(404)
            self.end_headers()
            return

        # Read body
        length = int(self.headers.get("Content-Length", 0))
        body = json.loads(self.rfile.read(length))
        message = body.get("message", "")
        if not message:
            self.send_response(422)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"ok": False, "error": "message required"}).encode())
            return

        # Run claude
        try:
            result = subprocess.run(
                ["claude", "-p", message, "--resume"],
                capture_output=True, text=True, timeout=120,
                cwd="C:\\Users\\raest\\Documents\\Karma_SADE"
            )
            response_text = result.stdout.strip() or result.stderr.strip() or "(no output)"
            self.send_response(200)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({
                "ok": True,
                "response": response_text,
                "exit_code": result.returncode
            }).encode())
        except subprocess.TimeoutExpired:
            self.send_response(504)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"ok": False, "error": "Claude timeout (120s)"}).encode())
        except Exception as e:
            self.send_response(500)
            self.send_header("Content-Type", "application/json")
            self.end_headers()
            self.wfile.write(json.dumps({"ok": False, "error": str(e)}).encode())

if __name__ == "__main__":
    print(f"[cc-server] Starting on port {PORT}")
    print(f"[cc-server] Auth: {'ENABLED' if TOKEN else 'DISABLED (set HUB_CHAT_TOKEN)'}")
    server = HTTPServer(("0.0.0.0", PORT), CCHandler)
    server.serve_forever()
```

**Step 2: Add /cc route to server.js**

After the existing routes (before `app.listen`), add:

```javascript
// P0N-A: CC proxy route — forwards to CC server on P1 via Tailscale
const CC_SERVER_URL = process.env.CC_SERVER_URL || 'http://100.124.194.102:7891';

app.post('/cc', requireAuth, async (req, res) => {
  const { message, session_id } = req.body;
  if (!message) {
    return res.status(422).json({ ok: false, error: 'message required' });
  }
  try {
    const response = await fetch(`${CC_SERVER_URL}/cc`, {
      method: 'POST',
      headers: {
        'Authorization': req.headers['authorization'],
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ message, session_id }),
      signal: AbortSignal.timeout(130000)  // 130s (CC server has 120s timeout)
    });
    const data = await response.json();
    res.status(response.status).json(data);
  } catch (err) {
    console.error('[/cc] proxy error:', err.message);
    res.status(502).json({ ok: false, error: `CC proxy error: ${err.message}` });
  }
});

app.get('/cc/health', async (req, res) => {
  try {
    const response = await fetch(`${CC_SERVER_URL}/health`, {
      signal: AbortSignal.timeout(5000)
    });
    const data = await response.json();
    res.json({ ok: true, cc_server: data });
  } catch (err) {
    res.status(502).json({ ok: false, error: `CC unreachable: ${err.message}` });
  }
});
```
